var arr = ["pig","cat","monkey"];//没有关键字时用中括号

for(var i=0;i<arr.length;i++>//使用数组关键字length应运而生了

if(arr[i]!="pig")console.log(arr[i]);//这里面相当于arr[0]输出的值，你的明白？
