arguments使用

初始化arguments

function fun(){
console.log(arguments[0]);
console.log(arguments[1]);
console.log(arguments[2]);
}
fun("第一个","第二个","第三个");

平方与乘积

function fun(){
if(arguments.length==1){
console.log(arguments[0]*arguments[0]);
}else if(arguments.length==2){
console.log(arguments[0]*arguments[1]);
}else{
console.log("超过位数");
}
}
fun(2,3);
